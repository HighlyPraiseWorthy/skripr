// In page.tsx — cookie-based A/B split
const bannerVariant = cookies().get('banner_v')?.value   // 'a' | 'b'

if (bannerVariant === 'b') {
  // Fyre red banner
  <TopBanner
    variant="fyre"
    headline="2 scripts free — no card required"
    cta="Claim 2 Free Scripts"
  />
} else {
  // Current Skripr violet banner
  <TopBanner
    variant="skripr"
    headline="2 scripts free — no card required"
    cta="Claim 2 Free Scripts"
  />
}

// Track conversion (send to Mixpanel / PostHog / Supabase)
track('banner_cta_clicked', { variant: bannerVariant })
